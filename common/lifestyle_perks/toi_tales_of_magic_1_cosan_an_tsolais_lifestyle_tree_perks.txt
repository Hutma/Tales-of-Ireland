learn_by_the_books_perk = {
	lifestyle = tales_of_magic_lifestyle
	tree = cosan_an_tsolais
	position = { 2 2 }
	icon = node_learning
	
	parent = fast_learner_perk
	
	effect = {
		if = {
			limit = { 
				AND = {
					NOT = { has_trait = mystic_1 }
					NOT = { has_trait = mystic_2 }
					NOT = { has_trait = mystic_3 }
				}
			}
			add_trait = mystic_1
		}
		else_if = {
			limit = { has_trait = mystic_1 }
			add_trait = mystic_2
		}
		else_if = {
			limit = { has_trait = mystic_2 }
			add_trait = mystic_3
		}
	}
	
	character_modifier = {
		learning = 3
	}
}

# On actions for the toi_otherworld_exploration activity

#To send an appropriate start event
otherworld_exploration_setup = {
	events = {
		toi_otherworld_exploration.1001 # Start of generic otherworld_exploration
	}
}

otherworld_exploration_activity_limit_hub = {
	effect = {
		activity = {
			remove_variable = is_having_otherworld_exploration_2001
			remove_variable = is_having_otherworld_exploration_2002
			remove_variable = is_having_otherworld_exploration_2003
			remove_variable = is_having_otherworld_exploration_2004
			remove_variable = is_having_otherworld_exploration_2005
			remove_variable = is_having_otherworld_exploration_2006
			remove_variable = is_having_otherworld_exploration_2007
		}
		if = {
			limit =  {
				exists = root.activity.var:number_of_places_counter
				OR = {
					root.activity.var:number_of_places_counter = 4
					root.activity.var:number_of_places_counter > 4
				}
			}
			trigger_event = {
				on_action = otherworld_exploration_return_events
				days = 5
			}
		}
		else = {
			trigger_event = {
				on_action = otherworld_exploration_place_choice_events
				days = 5
			}
			activity = {
				change_variable = {
					name = number_of_places_counter
					add = 1
				}
				if = { 
					limit = { exists = root.activity.var:number_of_core_events_counter }
					remove_variable = number_of_core_events_counter
				}
				set_variable = {
					name = number_of_core_events_counter
					value = 0
				}
			}
		}
	}
}

# On-action counting the core event for that place
otherworld_exploration_count_events = {
	effect = {
		if = {
			limit = {
				exists = root.activity.var:number_of_core_events_counter
				OR = {
					root.activity.var:number_of_core_events_counter = 4
					root.activity.var:number_of_core_events_counter > 4
				}
			}
			trigger_event = {
				on_action = otherworld_exploration_activity_limit_hub
				days = 5
			}
		}
		else_if = {
			limit = {
				exists = root.activity.var:number_of_core_events_counter
				root.activity.var:number_of_core_events_counter = 3
			}
			random_list = {
				25 = {
					trigger_event = {
						on_action = otherworld_exploration_core_events
						days = 1
					}
					activity = {
						change_variable = {
							name = number_of_core_events_counter
							add = 1
						}
					}
				}
				75 = {
					trigger_event = {
						on_action = otherworld_exploration_activity_limit_hub
						days = 5
					}
				}
			}
		}
		else_if = {
			limit = {
				exists = root.activity.var:number_of_core_events_counter
				root.activity.var:number_of_core_events_counter = 2
			}
			random_list = {
				60 = {
					trigger_event = {
						on_action = otherworld_exploration_core_events
						days = 1
					}
					activity = {
						change_variable = {
							name = number_of_core_events_counter
							add = 1
						}
					}
				}
				40 = {
					trigger_event = {
						on_action = otherworld_exploration_activity_limit_hub
						days = 5
					}
				}
			}				
		}
		else_if = {
			limit = {
				exists = root.activity.var:number_of_core_events_counter
				OR = {
					root.activity.var:number_of_core_events_counter = 1
					root.activity.var:number_of_core_events_counter = 0
				}
			}
			trigger_event = {
				on_action = otherworld_exploration_core_events
				days = 1
			}
			activity = {
				change_variable = {
					name = number_of_core_events_counter
					add = 1
				}
			}
		}
	}
}

GetSelectedArtifactTitle = {
	type = character
	first_valid = yes
	text = {
		localization_key = artifact_title
		trigger = {
			has_variable = artifact_selection
		}
	}
	text = {localization_key = artifact_name_fallback}
}

GetSelectedArtifactEquipped = {
	type = character
	# Equipped Artifacts
	text = {
		localization_key = equipped_artifact
		trigger = {
			has_variable = artifact_selection
			is_selection_equipped = yes
		}
	}
	
	# Unquipped Artifacts
	text = {
		localization_key = unequipped_artifact
		trigger = {
			exists = var:artifact_selection
			is_selection_equipped = no
		}
	}
	text = {localization_key = artifact_equip_fallback}
}

				every_held_title = {
					limit = {
						tier = tier_county
					}
					set_variable = {
						name = cattle_number
						value = this.development_level
					}
					change_variable = {
						name = cattle_number

					}
					remove_variable = cattle_number
				}
			}
		}
	}

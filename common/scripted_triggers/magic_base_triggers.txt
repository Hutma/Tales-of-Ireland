	OR = {
		var:dummy_char_target = {
			NOT = {
				has_character_flag = has_been_terrified
			}
		}
		var:dummy_char_target = {
			stress_level < 1
		}
		AND = {
			var:dummy_char_target = {
				stress_level > 1
			}
			OR = {
				piety_level >= var:dummy_char_target.piety_level
				piety_level >= 2
			}
		}
		AND = {
			var:dummy_char_target = {
				stress_level > 2
			}
			piety_level >= 4
		}
	}

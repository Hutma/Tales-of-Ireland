namespace = toi_magic_event

#Summon child spell
toi_magic_event.0001 = {
	title = toi_magic_event.0001.t
	desc = toi_magic_event.0001.desc
	type = character_event
	theme = family
	
	left_portrait = {
		character = root
		animation = admiration
	}
	
	right_portrait = {
		character = scope:sidhe_child
		animation = personality_rational
	}
	
	immediate = {
		if = {
			limit = {
				is_female = yes
			}
			create_character = {
				age = 1
				trait = magician_good_1
				random_traits = yes
				mother = root
				gender_female_chance = 50
				dynasty = root.dynasty
				faith = root.faith
				culture = root.culture
				employer = root
				save_scope_as = sidhe_child
			}
		}
		if = {
			limit = {
				is_female = no
			}
			create_character = {
				age = 1
				trait = magician_good_1
				random_traits = yes
				father = root
				gender_female_chance = 50
				dynasty = root.dynasty
				faith = root.faith
				culture = root.culture
				employer = root
				save_scope_as = sidhe_child
			}
		}
	}
	option = {
		name = toi_magic_event.0001.a
		remove_character_modifier = asked_sidhe_for_child
	}
}

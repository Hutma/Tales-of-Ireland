	}
	trigger = {
		
	}
	
	immediate = {
		create_character = {
			age = { 30 55 }

			random_traits = yes
			gender_female_chance = 0
			dynasty = none
			faith = faith:sidhe_beliefs
			culture = this.culture
			save_scope_as = giant
		}
	}
	
	option = {
		name = toi_decisions.004.a #I'll kill you, beast!
		duel = {
			skill = prowess
			value = 12
		
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				desc = toi_decisions.004.a.success
				send_interface_toast = {
					type = event_toast_effect_good
					title = toi_decisions.004.a.success
					scope:giant = {
						death = {
								death_reason = death_head_ripped_off
								killer = root
						}
					}
					add_prestige = 400
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				desc = toi_decisions.004.a.failure
				send_interface_toast = {
					type = event_toast_effect_bad
					title = toi_decisions.004.a.failure
					add_trait = wounded_2
					add_prestige = -100
				}
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				has_trait = brave
				add = 50
			}
			modifier = {
				has_trait = ambitious
				add = 20
			}
		}
	}
	option = {
		name = toi_decisions.004.b #Run for your life
		add_prestige = -200		
		ai_chance = {
			base = 90

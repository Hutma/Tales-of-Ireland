force_cattle_trade_outcome.0001 = {
	type = character_event
	title = force_cattle_trade_outcome.0001.t
	desc = force_cattle_trade_outcome.0001.desc
	theme = diplomacy_foreign_affairs_focus

	left_portrait = {
		character = root
		animation = scheme
	}
	left_portrait = {
		character = scope:target
		animation = personality_irrational
	}

	option = { # I will get 1 for free
		name = force_cattle_trade_outcome.0001.a
		scope:target = {
			capital_county = {
				change_development_level = -1
			}
		}

			}
		}

			change_development_level = 1
		}

			trigger_event = {
				id = force_cattle_trade_outcome.0002
				days = 1
			}
		}

		ai_chance = {
			base = 60
		}
	}

			scope:target = {

					development_level > 5

		}

		}

			base = 70
		}
	}

	option = { # get 10
		name = force_cattle_trade_outcome.0001.c

			}
		}
		
		root = {
			pay_short_term_gold = {

			}
		}
		capital_county = {
			change_development_level = 10
		}
